from rdkit import Chem
from rdkit.Chem import rdFingerprintGenerator

# Define two molecules: ethanol and dimethyl ether
ethanol_smiles = "CCO"  # Ethanol
dimethyl_ether_smiles = "COC"  # Dimethyl ether

ethanol = Chem.MolFromSmiles(ethanol_smiles)
dimethyl_ether = Chem.MolFromSmiles(dimethyl_ether_smiles)

# Generate Morgan fingerprints with a very small size to force a hash collision
fp_size = 1024  # Extremely small fingerprint size
radius = 2
fpgen = rdFingerprintGenerator.GetMorganGenerator(radius=radius, fpSize=fp_size)

ethanol_fp = fpgen.GetFingerprint(ethanol)
dimethyl_ether_fp = fpgen.GetFingerprint(dimethyl_ether)

# Compare fingerprints
common_bits = sum(1 for i in range(fp_size) if ethanol_fp.GetBit(i) and dimethyl_ether_fp.GetBit(i))
ethanol_bits = sum(1 for i in range(fp_size) if ethanol_fp.GetBit(i))
dimethyl_ether_bits = sum(1 for i in range(fp_size) if dimethyl_ether_fp.GetBit(i))

print(f"Ethanol fingerprint bits set: {ethanol_bits}")
print(f"Dimethyl ether fingerprint bits set: {dimethyl_ether_bits}")
print(f"Common bits: {common_bits}")
